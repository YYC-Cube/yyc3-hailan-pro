/**
 * AES配置

export interface AESConfig {
  /** 算法类型 */
  algorithm: 'aes-256-gcm' | 'aes-256-cbc' | 'aes-128-gcm';
  /** 密钥(32字节) */
  key: Uint8Array;
  /** 初始化向量(12-16字节) */
  iv?: Uint8Array;
  /** 认证标签(16字节，仅GCM模式) */
  authTag?: Uint8Array;
  /** 附加认证数据 */
  aad?: Uint8Array;
}

/**
 * AES加密结果
 */
export interface AESEncryptionResult {
  /** 加密数据 */
  ciphertext: Uint8Array;
  /** 初始化向量 */
  iv: Uint8Array;
  /** 认证标签(GCM模式) */
  authTag?: Uint8Array;
  /** 算法类型 */
  algorithm: string;
}

/**
 * ChaCha20配置
 */
export interface ChaCha20Config {
  /** 密钥(32字节) */
  key: Uint8Array;
  /** Nonce(12字节) */
  nonce: Uint8Array;
  /** 计数器初始值 */
  counter?: number;
  /** 附加认证数据 */
  aad?: Uint8Array;
}

/**
 * 对称加密输入
 */
export interface SymmetricEncryptInput {
  /** 待加密数据 */
  plaintext: string | Uint8Array;
  /** 加密配置 */
  config: AESConfig | ChaCha20Config;
  /** 加密级别 */
  level: EncryptionLevel;
}

/**
 * 对称解密输入
 */
export interface SymmetricDecryptInput {
  /** 待解密数据 */
  ciphertext: Uint8Array;
  /** 加密配置 */
  config: AESConfig | ChaCha20Config;
  /** 认证标签 */
  authTag?: Uint8Array;
}

/**
 * 对称加密服务接口
 */
export interface ISymmetricCryptoService {
  /**
   * 生成随机密钥

   * 生成随机IV

   * 加密数据

  exportKey(key: Uint8Array, format: 'raw' | 'hex' | 'base64'): string;
