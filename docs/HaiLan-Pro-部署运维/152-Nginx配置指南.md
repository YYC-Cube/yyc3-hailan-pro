### 3. Nginx配置指南

#### 3.1 基础配置

**nginx.conf：**

```nginx
user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log warn;
pid /var/run/nginx.pid;

events {
    worker_connections 1024;
    use epoll;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for"';

    access_log /var/log/nginx/access.log main;

    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    keepalive_timeout 65;
    types_hash_max_size 2048;

    # Gzip 压缩
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml text/javascript
               application/json application/javascript application/xml+rss;

    # 包含站点配置
    include /etc/nginx/conf.d/*.conf;
}
```

#### 3.2 HTTPS 配置

**SSL 终止配置：**

```nginx
server {
    listen 443 ssl http2;
    server_name hailan.pro;

    ssl_certificate /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;

    # HSTS
    add_header Strict-Transport-Security "max-age=63072000" always;

    # 安全头
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    location / {
        proxy_pass http://hailan-frontend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

# HTTP 重定向到 HTTPS
server {
    listen 80;
    server_name hailan.pro;
    return 301 https://$server_name$request_uri;
}
```

#### 3.3 负载均衡配置

**Upstream 配置：**

```nginx
upstream hailan-frontend {
    least_conn;
    server hailan-frontend-1:8080 max_fails=3 fail_timeout=30s;
    server hailan-frontend-2:8080 max_fails=3 fail_timeout=30s;
    server hailan-frontend-3:8080 max_fails=3 fail_timeout=30s;

    keepalive 32;
    keepalive_timeout 60s;
}

upstream hailan-backend {
    least_conn;
    server hailan-backend-1:8080 max_fails=3 fail_timeout=30s;
    server hailan-backend-2:8080 max_fails=3 fail_timeout=30s;
}
```
